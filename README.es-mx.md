
```bash

  --driver sqlite3 \
  --dburl ./exotic_birds.db \

  --outdir ./schemas
```

```bash

  --format yaml \
  --outdir ./schemas
```

###



```golang
package main

import (
    "fmt"
	"github.com/tgallant/db2jsonschema"
)

func main() {
  request := &db2jsonschema.Request{
      Driver:     "sqlite3",
      DataSource: "./test.db",
      Format:     "yaml",
      Outdir:     "./schemas",
  }
  err := request.Perform()
  if err != nil {
      fmt.Println(err)
  }
}
```
